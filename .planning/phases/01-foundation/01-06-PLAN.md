Create `middleware.ts` in the project root (NOT in `src/`) with the correct Clerk Next.js 15 pattern:

```typescript
import { clerkMiddleware, createRouteMatcher } from '@clerk/nextjs/server'

// Public routes — accessible without authentication
const isPublicRoute = createRouteMatcher([
  '/',
  '/sign-in(.*)',
  '/sign-up(.*)',
  '/compare(.*)',
  '/firms(.*)',
  '/guides(.*)',
  '/rewards',
  '/go(.*)',          // affiliate redirect — must be public
  '/legal(.*)',
  '/api/webhooks(.*)', // Clerk webhook — must be public
])

// Admin routes — require admin role in Clerk publicMetadata
const isAdminRoute = createRouteMatcher([
  '/admin(.*)',
])

export default clerkMiddleware(async (auth, request) => {
  // Admin routes: require authentication AND admin role
  if (isAdminRoute(request)) {
    const { userId, sessionClaims } = await auth()
    if (!userId) {
      const { redirectToSignIn } = await import('@clerk/nextjs/server')
      return redirectToSignIn({ returnBackUrl: request.url })
    }
    // Check admin role in publicMetadata
    const role = (sessionClaims?.metadata as { role?: string })?.role
    if (role !== 'admin') {
      const { NextResponse } = await import('next/server')
      return NextResponse.redirect(new URL('/', request.url))
    }
  }

  // Protected routes (not public): require authentication
  if (!isPublicRoute(request)) {
    await auth.protect()
  }
})

export const config = {
  matcher: [
    // Skip Next.js internals and static files
    '/((?!_next|[^?]*\\.(?:html?|css|js(?!on)|jpe?g|webp|png|gif|svg|ttf|woff2?|ico|csv|docx?|xlsx?|zip|webmanifest)).*)',
    // Always run for API routes
    '/(api|trpc)(.*)',
  ],
}
```

<threat_model>
## Security Threat Model

| Threat | Severity | Mitigation |
|--------|----------|-----------|
| Admin route bypass via middleware miss | High | `isAdminRoute` + role check runs in middleware before any page code executes |
| Clerk deprecated `withClerk` wrapper used | High | Uses `clerkMiddleware()` — the only supported Next.js 15 pattern |
| `auth.protect()` throwing uncaught errors on protected routes | Medium | Clerk's `auth.protect()` automatically redirects to sign-in — no try/catch needed |
| Webhook endpoint requiring auth (breaks Clerk webhook calls) | High | `/api/webhooks(.*)` is in `isPublicRoute` — Clerk webhook calls are unauthenticated by design, verified by signature instead |
| CSRF on auth forms | Low | Clerk handles CSRF internally via its session tokens and SameSite cookie policies |
</threat_model>
