// PropPilot Database Schema
// All 14 models for v1.0

generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider  = "postgresql"
  url       = env("DATABASE_URL")
  directUrl = env("DIRECT_URL")
}

// ─── Enums ───────────────────────────────────────────────────────────────

enum ClaimStatus {
  PENDING
  APPROVED
  REJECTED
}

enum RewardTierName {
  BRONZE
  SILVER
  GOLD
  PLATINUM
}

enum DrawdownType {
  BALANCE
  EQUITY
  TRAILING_BALANCE
  TRAILING_EQUITY
}

enum EvaluationType {
  ONE_STEP
  TWO_STEP
  THREE_STEP
  INSTANT_FUNDING
  FUNDED_DIRECT
}

enum FundingStyle {
  CHALLENGE
  INSTANT
  FUNDED
}

// ─── User ────────────────────────────────────────────────────────────────

model User {
  id        String   @id @default(cuid())
  clerkId   String   @unique
  email     String   @unique
  firstName String?
  lastName  String?
  imageUrl  String?
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  // Relations
  rewardPoints    UserRewardPoints?
  pointTxns       PointTransaction[]
  purchaseClaims  PurchaseClaim[]
  savedFirms      SavedFirm[]
  reviews         Review[]
  affiliateClicks AffiliateClick[]
  copyEvents      CopyEvent[]

  @@index([clerkId])
  @@index([email])
}

// ─── Firm ────────────────────────────────────────────────────────────────

model Firm {
  id              String   @id @default(cuid())
  slug            String   @unique
  name            String
  logoUrl         String?
  description     String?
  websiteUrl      String?
  affiliateUrl    String?
  isActive        Boolean  @default(true)
  isFeatured      Boolean  @default(false)
  isVerified      Boolean  @default(false)
  lastVerifiedAt  DateTime?
  createdAt       DateTime @default(now())
  updatedAt       DateTime @updatedAt

  // Metrics
  maxAccountSize      Int?           // in USD
  minAccountSize      Int?           // in USD
  challengeFee        Decimal?       @db.Decimal(10, 2)
  profitTarget        Float?         // percentage, e.g. 10.0
  dailyLossLimit      Float?         // percentage, e.g. 5.0
  maxDrawdown         Float?         // percentage, e.g. 10.0
  drawdownType        DrawdownType?
  evaluationType      EvaluationType?
  fundingStyle        FundingStyle?
  payoutSpeed         String?        // e.g. "1-3 business days"
  profitSplit         Float?         // percentage, e.g. 80.0
  minTradingDays      Int?
  consistencyRule     Boolean        @default(false)
  consistencyRuleNote String?
  newsTrading         Boolean        @default(false)
  weekendHolding      Boolean        @default(false)
  eaAllowed           Boolean        @default(false)
  scalingPlan         Boolean        @default(false)
  platforms           String[]       // e.g. ["MT4", "MT5", "DXTrade"]

  // Scoring
  ruleDifficultyScore Int?           // 0–100, computed by admin

  // Relations
  offers          FirmOffer[]
  affiliateClicks AffiliateClick[]
  copyEvents      CopyEvent[]
  purchaseClaims  PurchaseClaim[]
  savedFirms      SavedFirm[]
  reviews         Review[]
  faqs            FAQ[]
  stats           ComparisonStats?

  @@index([slug])
  @@index([isActive, isFeatured])
  @@index([drawdownType])
  @@index([evaluationType])
}

// ─── FirmOffer (Discount Code) ────────────────────────────────────────────

model FirmOffer {
  id              String   @id @default(cuid())
  firmId          String
  code            String
  discountPercent Float?   // e.g. 15.0 for 15% off
  discountAmount  Decimal? @db.Decimal(10, 2) // fixed amount off
  description     String?
  isExclusive     Boolean  @default(false)
  isActive        Boolean  @default(true)
  affiliateUrl    String?  // override firm's default affiliate URL
  expiresAt       DateTime?
  createdAt       DateTime @default(now())
  updatedAt       DateTime @updatedAt

  // Relations
  firm            Firm       @relation(fields: [firmId], references: [id], onDelete: Cascade)
  affiliateClicks AffiliateClick[]
  copyEvents      CopyEvent[]
  purchaseClaims  PurchaseClaim[]

  @@index([firmId, isActive])
  @@index([code])
}

// ─── AffiliateClick ──────────────────────────────────────────────────────

model AffiliateClick {
  id          String   @id @default(cuid())
  firmId      String
  offerId     String?
  userId      String?
  sessionId   String?
  referrer    String?
  userAgent   String?
  ipHash      String?  // hashed for privacy
  createdAt   DateTime @default(now())

  // Relations
  firm        Firm       @relation(fields: [firmId], references: [id], onDelete: Cascade)
  offer       FirmOffer? @relation(fields: [offerId], references: [id], onDelete: SetNull)
  user        User?      @relation(fields: [userId], references: [id], onDelete: SetNull)

  @@index([firmId, createdAt])
  @@index([userId])
  @@index([createdAt])
}

// ─── CopyEvent ───────────────────────────────────────────────────────────

model CopyEvent {
  id        String   @id @default(cuid())
  firmId    String
  offerId   String?
  userId    String?
  sessionId String?
  createdAt DateTime @default(now())

  // Relations
  firm      Firm       @relation(fields: [firmId], references: [id], onDelete: Cascade)
  offer     FirmOffer? @relation(fields: [offerId], references: [id], onDelete: SetNull)
  user      User?      @relation(fields: [userId], references: [id], onDelete: SetNull)

  @@index([firmId, createdAt])
  @@index([userId])
}

// ─── PurchaseClaim ────────────────────────────────────────────────────────

model PurchaseClaim {
  id             String      @id @default(cuid())
  userId         String
  firmId         String
  offerId        String?
  orderId        String?
  purchaseEmail  String
  amountPaid     Decimal     @db.Decimal(10, 2)
  purchaseDate   DateTime
  discountCode   String?
  receiptUrl     String      // UploadThing URL
  status         ClaimStatus @default(PENDING)
  rejectionReason String?
  pointsAwarded  Int?
  adminNotes     String?
  reviewedAt     DateTime?
  reviewedBy     String?     // admin user ID
  createdAt      DateTime    @default(now())
  updatedAt      DateTime    @updatedAt

  // Relations
  user           User       @relation(fields: [userId], references: [id], onDelete: Cascade)
  firm           Firm       @relation(fields: [firmId], references: [id], onDelete: Restrict)
  offer          FirmOffer? @relation(fields: [offerId], references: [id], onDelete: SetNull)

  @@index([userId, status])
  @@index([firmId])
  @@index([status, createdAt])
}

// ─── UserRewardPoints ─────────────────────────────────────────────────────

model UserRewardPoints {
  id        String         @id @default(cuid())
  userId    String         @unique
  balance   Int            @default(0)
  tier      RewardTierName @default(BRONZE)
  createdAt DateTime       @default(now())
  updatedAt DateTime       @updatedAt

  // Relations
  user      User           @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@index([userId])
  @@index([tier])
}

// ─── PointTransaction ─────────────────────────────────────────────────────

model PointTransaction {
  id          String   @id @default(cuid())
  userId      String
  points      Int      // positive = earn, negative = redeem
  description String
  claimId     String?  // reference to PurchaseClaim if applicable
  createdAt   DateTime @default(now())

  // Relations
  user        User     @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@index([userId, createdAt])
}

// ─── RewardTier ──────────────────────────────────────────────────────────

model RewardTier {
  id              String         @id @default(cuid())
  name            RewardTierName @unique
  minPoints       Int
  maxPoints       Int?
  pointsPerDollar Int            @default(1)
  benefits        String[]
  badgeColor      String?        // hex color
  createdAt       DateTime       @default(now())
  updatedAt       DateTime       @updatedAt

  @@index([minPoints])
}

// ─── Review ───────────────────────────────────────────────────────────────

model Review {
  id          String   @id @default(cuid())
  userId      String
  firmId      String
  rating      Int      // 1–5
  title       String?
  content     String
  isApproved  Boolean  @default(false)
  isVerified  Boolean  @default(false) // verified purchase
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

  // Relations
  user        User     @relation(fields: [userId], references: [id], onDelete: Cascade)
  firm        Firm     @relation(fields: [firmId], references: [id], onDelete: Cascade)

  @@unique([userId, firmId]) // one review per user per firm
  @@index([firmId, isApproved])
  @@index([userId])
}

// ─── GuidePost ────────────────────────────────────────────────────────────

model GuidePost {
  id          String   @id @default(cuid())
  slug        String   @unique
  title       String
  excerpt     String?
  content     String   // rich text / markdown
  category    String?
  tags        String[]
  isPublished Boolean  @default(false)
  publishedAt DateTime?
  authorName  String?
  readingTime Int?     // minutes
  metaTitle   String?
  metaDesc    String?
  ogImage     String?
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

  @@index([slug])
  @@index([isPublished, publishedAt])
  @@index([category])
}

// ─── FAQ ─────────────────────────────────────────────────────────────────

model FAQ {
  id        String   @id @default(cuid())
  firmId    String?  // null = global FAQ
  question  String
  answer    String
  sortOrder Int      @default(0)
  isActive  Boolean  @default(true)
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  // Relations
  firm      Firm?    @relation(fields: [firmId], references: [id], onDelete: Cascade)

  @@index([firmId, sortOrder])
}

// ─── AdminActionLog ───────────────────────────────────────────────────────

model AdminActionLog {
  id         String   @id @default(cuid())
  adminId    String   // Clerk user ID of admin
  action     String   // e.g. "APPROVE_CLAIM", "REJECT_CLAIM", "UPDATE_FIRM"
  targetType String   // e.g. "PurchaseClaim", "Firm", "FirmOffer"
  targetId   String   // ID of the affected record
  metadata   Json?    // additional context (rejection reason, old values, etc.)
  createdAt  DateTime @default(now())

  @@index([adminId])
  @@index([targetType, targetId])
  @@index([createdAt])
}

// ─── SavedFirm ────────────────────────────────────────────────────────────

model SavedFirm {
  id        String   @id @default(cuid())
  userId    String
  firmId    String
  createdAt DateTime @default(now())

  // Relations
  user      User     @relation(fields: [userId], references: [id], onDelete: Cascade)
  firm      Firm     @relation(fields: [firmId], references: [id], onDelete: Cascade)

  @@unique([userId, firmId])
  @@index([userId])
}

// ─── ComparisonStats ─────────────────────────────────────────────────────

model ComparisonStats {
  id         String   @id @default(cuid())
  firmId     String   @unique
  pageViews  Int      @default(0)
  weekViews  Int      @default(0)
  createdAt  DateTime @default(now())
  updatedAt  DateTime @updatedAt

  // Relations
  firm       Firm     @relation(fields: [firmId], references: [id], onDelete: Cascade)

  @@index([firmId])
}

<threat_model>
## Security Threat Model

| Threat | Severity | Mitigation |
|--------|----------|-----------|
| IP address stored in AffiliateClick | Medium | `ipHash` field stores hashed IP only — never the raw IP address |
| Cascade delete removing user's claim history | Medium | PurchaseClaim uses `onDelete: Cascade` on User (intentional — GDPR compliance) |
| Missing unique constraints allowing duplicate reviews | Low | `@@unique([userId, firmId])` prevents duplicate reviews per user per firm |
| Admin action log not capturing sensitive data | Low | `metadata Json?` field stores context without exposing PII — admin ID stored, not full user data |
</threat_model>
