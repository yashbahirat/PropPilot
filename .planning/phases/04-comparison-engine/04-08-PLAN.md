```typescript
'use client';

import { NuqsAdapter } from 'nuqs/adapters/next/app';
import { ScoredFirm } from '@/lib/scoring';
import { FilterBar } from './FilterBar';
import { FilterDrawer } from './FilterDrawer';
import { SearchBar } from './SearchBar';
import { SortControls } from './SortControls';
import { FirmList } from './FirmList';
import { ComparisonDock } from './ComparisonDock';
import { ComparisonView } from './ComparisonView';
import { AffiliateDisclosureBanner } from './AffiliateDisclosureBanner';
import { useCompareParams } from '@/hooks/use-compare-params';

interface ComparePageInnerProps {
  firms: ScoredFirm[];
}

function ComparePageInner({ firms }: ComparePageInnerProps) {
  const { params } = useCompareParams();
  const isComparisonView = params.compare.length >= 2;

  // Firms selected for side-by-side comparison
  const comparedFirms = params.compare
    .map((slug) => firms.find((f) => f.slug === slug))
    .filter((f): f is ScoredFirm => Boolean(f));

  return (
    <div className="min-h-screen bg-[#08080F]">
      {/* Page Header */}
      <div className="max-w-7xl mx-auto px-4 pt-8 pb-4">
        <div className="flex flex-col gap-1 mb-6">
          <h1 className="text-2xl font-bold text-white">Compare Prop Firms</h1>
          <p className="text-sm text-white/50">
            {firms.length} firms • verified data • live discount codes
          </p>
        </div>

        {/* Affiliate Disclosure */}
        <AffiliateDisclosureBanner />
      </div>

      {/* Desktop Filter Bar — sticky */}
      <div className="hidden md:block">
        <FilterBar />
      </div>

      {/* Main Content Area */}
      <div className="max-w-7xl mx-auto px-4 py-4">
        {/* Search + Sort + Mobile Filter Button */}
        <div className="flex items-center gap-3 mb-4">
          <div className="flex-1 max-w-xs">
            <SearchBar />
          </div>
          {/* Mobile filter trigger */}
          <div className="md:hidden">
            <FilterDrawer />
          </div>
          <div className="ml-auto hidden md:block">
            <SortControls />
          </div>
        </div>

        {/* Mobile Sort Controls */}
        <div className="md:hidden mb-4">
          <SortControls />
        </div>

        {/* Comparison View (full-page) */}
        {isComparisonView && comparedFirms.length >= 2 ? (
          <div className="bg-[#0E0E1A] rounded-2xl border border-white/10 overflow-hidden">
            <div className="px-6 py-4 border-b border-white/5 flex items-center justify-between">
              <div>
                <h2 className="text-base font-semibold text-white">Side-by-Side Comparison</h2>
                <p className="text-xs text-white/40 mt-0.5">
                  Differences highlighted in amber • identical values shown normally
                </p>
              </div>
              <button
                onClick={() => {
                  // Clear compare param — handled by useCompareParams
                  window.history.pushState({}, '', '/compare');
                  window.location.reload(); // simplest way to clear nuqs state
                }}
                className="text-xs text-white/40 hover:text-white/70 transition-colors"
              >
                ← Back to list
              </button>
            </div>
            <ComparisonView firms={comparedFirms} />
          </div>
        ) : (
          /* Firm List */
          <div className="bg-[#0E0E1A] rounded-2xl border border-white/10 overflow-hidden">
            <FirmList firms={firms} />
          </div>
        )}
      </div>

      {/* Comparison Dock (sticky bottom) */}
      <ComparisonDock firms={firms} />

      {/* Bottom padding for dock */}
      <div className="h-24" />
    </div>
  );
}

interface ComparePageClientProps {
  firms: ScoredFirm[];
}

export function ComparePageClient({ firms }: ComparePageClientProps) {
  return (
    <NuqsAdapter>
      <ComparePageInner firms={firms} />
    </NuqsAdapter>
  );
}
```

Note: `NuqsAdapter` import path may differ in nuqs v2. Check `node_modules/nuqs/dist/index.d.ts` or `node_modules/nuqs/adapters/` for the correct adapter for Next.js App Router. Common paths:
- `nuqs/adapters/next/app` (nuqs v2+)
- `nuqs` directly (if NuqsAdapter is re-exported from main)

If adapter not needed (nuqs v2 auto-detects Next.js App Router), remove the NuqsAdapter wrapper and just render ComparePageInner directly.
