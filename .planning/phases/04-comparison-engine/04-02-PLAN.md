## Objective

Define the canonical nuqs URL parameter schema for the entire compare page — all filters, sort, search, and comparison slugs — in a single shared module consumed by all client components.

<read_first>

<action>
Create `src/lib/compare-params.ts`:

```typescript
import {
  parseAsArrayOf,
  parseAsBoolean,
  parseAsFloat,
  parseAsString,
  parseAsStringEnum,
  createSearchParamsCache,
} from 'nuqs/server';

// ─── Sort Options ───────────────────────────────────────────────────────────

export const SortOption = {
  SCORE: 'score',
  COST: 'cost',
  DISCOUNT: 'discount',
  DRAWDOWN: 'drawdown',
  PAYOUT: 'payout',
} as const;

export type SortOption = (typeof SortOption)[keyof typeof SortOption];

// ─── Param Parsers ──────────────────────────────────────────────────────────

export const compareParamsParsers = {
  // Search
  q: parseAsString.withDefault(''),

  // Sort
  sort: parseAsStringEnum(Object.values(SortOption)).withDefault(SortOption.SCORE),

  // Filters — string enums matching Prisma enum values
  drawdownType: parseAsStringEnum([
    'BALANCE',
    'EQUITY',
    'TRAILING_BALANCE',
    'TRAILING_EQUITY',
  ] as const).withDefault('' as ''),

  evaluationType: parseAsStringEnum([
    'ONE_STEP',
    'TWO_STEP',
    'THREE_STEP',
    'INSTANT_FUNDING',
    'FUNDED_DIRECT',
  ] as const).withDefault('' as ''),

  fundingStyle: parseAsStringEnum([
    'CHALLENGE',
    'INSTANT',
    'FUNDED',
  ] as const).withDefault('' as ''),

  // Fee range
  minFee: parseAsFloat.withDefault(0),
  maxFee: parseAsFloat.withDefault(0), // 0 means no upper limit

  // Boolean filters
  hasDiscount: parseAsBoolean.withDefault(false),
  newsTrading: parseAsBoolean.withDefault(false),
  weekendHolding: parseAsBoolean.withDefault(false),
  eaAllowed: parseAsBoolean.withDefault(false),

  // Side-by-side comparison (up to 3 firm slugs)
  compare: parseAsArrayOf(parseAsString).withDefault([]),
};

// ─── Server-Side Cache (for Server Components if needed) ───────────────────

export const compareParamsCache = createSearchParamsCache(compareParamsParsers);

// ─── Types ──────────────────────────────────────────────────────────────────

export type CompareSearchParams = {
  [K in keyof typeof compareParamsParsers]: ReturnType<
    (typeof compareParamsParsers)[K]['parseServerSide']
  >;
};
```

Note: nuqs v2 uses `parseAsStringEnum` which may require an array (not `as const` enum). Check the exact nuqs v2 API in `node_modules/nuqs/dist/index.d.ts` — adjust if `parseAsStringEnum` signature differs. The fallback is using `parseAsString` for enum fields and validating in the filter logic.
</action>

### Task 2: Create Client-Side useCompareParams Hook

<read_first>
- src/lib/compare-params.ts (parsers just created)
- node_modules/nuqs/dist/index.d.ts (useQueryStates signature)
</read_first>

<action>
Create `src/hooks/use-compare-params.ts`:

```typescript
'use client';

import { useQueryStates } from 'nuqs';
import { compareParamsParsers } from '@/lib/compare-params';

export function useCompareParams() {
  const [params, setParams] = useQueryStates(compareParamsParsers, {
    shallow: true, // Don't trigger full page re-render on filter changes
    history: 'replace', // Replace instead of push to avoid polluting browser history
  });

  const clearFilter = (key: keyof typeof compareParamsParsers) => {
    setParams({ [key]: null } as Parameters<typeof setParams>[0]);
  };

  const clearAllFilters = () => {
    setParams({
      q: null,
      sort: null,
      drawdownType: null,
      evaluationType: null,
      fundingStyle: null,
      minFee: null,
      maxFee: null,
      hasDiscount: null,
      newsTrading: null,
      weekendHolding: null,
      eaAllowed: null,
      compare: null,
    });
  };

  const addToCompare = (slug: string) => {
    if (params.compare.length >= 3 || params.compare.includes(slug)) return;
    setParams({ compare: [...params.compare, slug] });
  };

  const removeFromCompare = (slug: string) => {
    setParams({ compare: params.compare.filter((s) => s !== slug) });
  };

  const isInCompare = (slug: string) => params.compare.includes(slug);

  const hasActiveFilters =
    params.q !== '' ||
    params.drawdownType !== '' ||
    params.evaluationType !== '' ||
    params.fundingStyle !== '' ||
    params.minFee > 0 ||
    params.maxFee > 0 ||
    params.hasDiscount ||
    params.newsTrading ||
    params.weekendHolding ||
    params.eaAllowed;

  return {
    params,
    setParams,
    clearFilter,
    clearAllFilters,
    addToCompare,
    removeFromCompare,
    isInCompare,
    hasActiveFilters,
  };
}
```
