## Objective

Build the sticky comparison dock bar (slides in from bottom when firms are selected, shows thumbnails + Compare button) and the full-page comparison table view (accessed via `?compare=slug1,slug2`, highlights differing cells in amber/teal, identical values in normal style).

```typescript
'use client';

import { LazyMotion, domAnimation, m, AnimatePresence } from 'framer-motion';
import { useCompareParams } from '@/hooks/use-compare-params';
import { Button } from '@/components/ui/button';
import { X, GitCompareArrows } from 'lucide-react';
import Link from 'next/link';
import { ScoredFirm } from '@/lib/scoring';
import Image from 'next/image';

interface ComparisonDockProps {
  firms: ScoredFirm[];
}

export function ComparisonDock({ firms }: ComparisonDockProps) {
  const { params, removeFromCompare } = useCompareParams();
  const selectedSlugs = params.compare;
  const selectedFirms = selectedSlugs
    .map((slug) => firms.find((f) => f.slug === slug))
    .filter((f): f is ScoredFirm => Boolean(f));

  const compareHref = selectedSlugs.length >= 2
    ? `/compare?compare=${selectedSlugs.join(',')}`
    : '#';

  return (
    <LazyMotion features={domAnimation}>
      <AnimatePresence>
        {selectedFirms.length > 0 && (
          <m.div
            key="comparison-dock"
            initial={{ y: 100, opacity: 0 }}
            animate={{ y: 0, opacity: 1 }}
            exit={{ y: 100, opacity: 0 }}
            transition={{ type: 'spring', stiffness: 400, damping: 35 }}
            className="fixed bottom-0 left-0 right-0 z-50 bg-[#0E0E1A]/95 backdrop-blur border-t border-white/10 shadow-2xl shadow-black/50"
          >
            <div className="max-w-7xl mx-auto px-4 py-3 flex items-center gap-4">
              {/* Selected firms */}
              <div className="flex items-center gap-3 flex-1 overflow-x-auto scrollbar-hide">
                {selectedFirms.map((firm) => (
                  <div
                    key={firm.slug}
                    className="flex items-center gap-2 bg-white/5 rounded-lg px-2.5 py-1.5 border border-white/10 shrink-0"
                  >
                    {/* Logo */}
                    <div className="h-6 w-6 rounded bg-white/5 flex items-center justify-center overflow-hidden">
                      {firm.logoUrl ? (
                        <Image src={firm.logoUrl} alt={firm.name} width={24} height={24} className="object-contain" />
                      ) : (
                        <span className="text-[10px] font-bold text-white/40">{firm.name.charAt(0)}</span>
                      )}
                    </div>
                    <span className="text-sm font-medium text-white whitespace-nowrap">{firm.name}</span>
                    <button
                      onClick={() => removeFromCompare(firm.slug)}
                      className="text-white/30 hover:text-white/70 transition-colors"
                      aria-label={`Remove ${firm.name} from comparison`}
                    >
                      <X className="h-3.5 w-3.5" />
                    </button>
                  </div>
                ))}

                {/* Slots */}
                {Array.from({ length: Math.max(0, 2 - selectedFirms.length) }).map((_, i) => (
                  <div
                    key={`slot-${i}`}
                    className="flex items-center gap-2 bg-white/[0.03] rounded-lg px-3 py-1.5 border border-dashed border-white/10 shrink-0"
                  >
                    <span className="text-sm text-white/20">Select a firm...</span>
                  </div>
                ))}
              </div>

              {/* Hint */}
              <p className="text-xs text-white/30 hidden md:block shrink-0">
                {selectedFirms.length}/3 selected
              </p>

              {/* Compare button */}
              <Button
                asChild={selectedSlugs.length >= 2}
                disabled={selectedSlugs.length < 2}
                className="shrink-0 bg-[#00D4AA] hover:bg-[#00D4AA]/90 text-[#08080F] font-semibold disabled:opacity-40 disabled:cursor-not-allowed"
              >
                {selectedSlugs.length >= 2 ? (
                  <Link href={compareHref}>
                    <GitCompareArrows className="h-4 w-4 mr-2" />
                    Compare ({selectedFirms.length})
                  </Link>
                ) : (
                  <span>
                    <GitCompareArrows className="h-4 w-4 mr-2" />
                    Compare ({selectedFirms.length})
                  </span>
                )}
              </Button>
            </div>
          </m.div>
        )}
      </AnimatePresence>
    </LazyMotion>
  );
}
```

```typescript
'use client';

import { ScoredFirm } from '@/lib/scoring';
import { ScoreBadge } from './ScoreBadge';
import { CopyCodeButton } from './CopyCodeButton';
import Image from 'next/image';
import Link from 'next/link';
import { ExternalLink, X } from 'lucide-react';
import { Button } from '@/components/ui/button';
import { useCompareParams } from '@/hooks/use-compare-params';
import { format } from 'date-fns';

interface ComparisonViewProps {
  firms: ScoredFirm[]; // pre-filtered to only the firms being compared
}

// A "row" in the comparison table
interface ComparisonRow {
  label: string;
  key: keyof ScoredFirm | 'bestOfferCode' | 'trueCost';
  format?: (firm: ScoredFirm) => React.ReactNode;
}

const COMPARISON_ROWS: ComparisonRow[] = [
  {
    label: 'Overall Score',
    key: 'overallScore',
    format: (f) => <ScoreBadge score={f.overallScore} size="sm" />,
  },
  {
    label: 'Challenge Fee',
    key: 'challengeFee',
    format: (f) => f.challengeFee ? `$${parseFloat(f.challengeFee).toFixed(0)}` : '—',
  },
  {
    label: 'True Cost (after discount)',
    key: 'trueCost',
    format: (f) => {
      if (!f.challengeFee) return '—';
      const base = parseFloat(f.challengeFee);
      const disc = f.bestOffer?.discountPercent;
      return disc ? `$${(base * (1 - disc / 100)).toFixed(0)}` : `$${base.toFixed(0)}`;
    },
  },
  {
    label: 'Discount Code',
    key: 'bestOfferCode',
    format: (f) =>
      f.bestOffer ? (
        <CopyCodeButton code={f.bestOffer.code} discountPercent={f.bestOffer.discountPercent} compact />
      ) : (
        <span className="text-white/30 text-xs">No discount</span>
      ),
  },
  {
    label: 'Max Drawdown',
    key: 'maxDrawdown',
    format: (f) => f.maxDrawdown ? `${f.maxDrawdown}%` : '—',
  },
  {
    label: 'Drawdown Type',
    key: 'drawdownType',
    format: (f) => {
      const labels: Record<string, string> = {
        BALANCE: 'Balance',
        EQUITY: 'Equity',
        TRAILING_BALANCE: 'Trailing Balance',
        TRAILING_EQUITY: 'Trailing Equity',
      };
      return f.drawdownType ? labels[f.drawdownType] ?? f.drawdownType : '—';
    },
  },
  {
    label: 'Daily Loss Limit',
    key: 'dailyLossLimit',
    format: (f) => f.dailyLossLimit ? `${f.dailyLossLimit}%` : '—',
  },
  {
    label: 'Profit Target',
    key: 'profitTarget',
    format: (f) => f.profitTarget ? `${f.profitTarget}%` : '—',
  },
  {
    label: 'Profit Split',
    key: 'profitSplit',
    format: (f) => f.profitSplit ? `${f.profitSplit}%` : '—',
  },
  {
    label: 'Payout Speed',
    key: 'payoutSpeed',
    format: (f) => f.payoutSpeed ?? '—',
  },
  {
    label: 'Evaluation Type',
    key: 'evaluationType',
    format: (f) =>
      f.evaluationType
        ? f.evaluationType.replace(/_/g, ' ').toLowerCase().replace(/\b\w/g, (c) => c.toUpperCase())
        : '—',
  },
  {
    label: 'News Trading',
    key: 'newsTrading',
    format: (f) => (
      <span className={f.newsTrading ? 'text-emerald-400' : 'text-red-400/70'}>
        {f.newsTrading ? '✓ Allowed' : '✗ Not allowed'}
      </span>
    ),
  },
  {
    label: 'Weekend Holding',
    key: 'weekendHolding',
    format: (f) => (
      <span className={f.weekendHolding ? 'text-emerald-400' : 'text-red-400/70'}>
        {f.weekendHolding ? '✓ Allowed' : '✗ Not allowed'}
      </span>
    ),
  },
  {
    label: 'EA / Bots Allowed',
    key: 'eaAllowed',
    format: (f) => (
      <span className={f.eaAllowed ? 'text-emerald-400' : 'text-red-400/70'}>
        {f.eaAllowed ? '✓ Allowed' : '✗ Not allowed'}
      </span>
    ),
  },
  {
    label: 'Consistency Rule',
    key: 'consistencyRule',
    format: (f) => (
      <span className={f.consistencyRule ? 'text-amber-400' : 'text-emerald-400'}>
        {f.consistencyRule ? '⚠ Required' : '✓ Not required'}
      </span>
    ),
  },
  {
    label: 'Last Verified',
    key: 'lastVerifiedAt',
    format: (f) =>
      f.lastVerifiedAt ? format(new Date(f.lastVerifiedAt), 'MMM d, yyyy') : '—',
  },
];

// Check if all values in a row are identical (stringify for comparison)
function rowIsDifferent(firms: ScoredFirm[], row: ComparisonRow): boolean {
  const values = firms.map((f) => {
    const val = f[row.key as keyof ScoredFirm];
    return typeof val === 'object' ? JSON.stringify(val) : String(val ?? '');
  });
  return new Set(values).size > 1;
}

export function ComparisonView({ firms }: ComparisonViewProps) {
  const { removeFromCompare } = useCompareParams();

  if (firms.length < 2) {
    return (
      <div className="text-center py-24 text-white/40">
        Select at least 2 firms to compare.
      </div>
    );
  }

  return (
    <div className="overflow-x-auto">
      <table className="w-full min-w-[600px]">
        <thead>
          <tr>
            <th className="text-left px-4 py-3 text-xs font-semibold text-white/30 uppercase tracking-wider w-40 border-b border-white/5">
              Metric
            </th>
            {firms.map((firm) => (
              <th key={firm.id} className="px-4 py-3 border-b border-white/5 text-center">
                <div className="flex flex-col items-center gap-2">
                  {/* Logo */}
                  <div className="h-10 w-10 rounded-xl bg-white/5 border border-white/10 flex items-center justify-center overflow-hidden">
                    {firm.logoUrl ? (
                      <Image src={firm.logoUrl} alt={firm.name} width={40} height={40} className="object-contain" />
                    ) : (
                      <span className="text-sm font-bold text-white/40">{firm.name.charAt(0)}</span>
                    )}
                  </div>
                  <Link href={`/firms/${firm.slug}`} className="text-sm font-semibold text-white hover:text-[#00D4AA] transition-colors">
                    {firm.name}
                  </Link>
                  <button
                    onClick={() => removeFromCompare(firm.slug)}
                    className="text-white/20 hover:text-white/50 transition-colors"
                    title="Remove from comparison"
                  >
                    <X className="h-3.5 w-3.5" />
                  </button>
                </div>
              </th>
            ))}
          </tr>
        </thead>
        <tbody>
          {COMPARISON_ROWS.map((row) => {
            const isDiff = rowIsDifferent(firms, row);
            return (
              <tr
                key={row.label}
                className={`border-b border-white/5 ${isDiff ? 'bg-amber-500/[0.03]' : ''}`}
              >
                <td className="px-4 py-3 text-xs font-medium text-white/40 whitespace-nowrap">
                  {row.label}
                  {isDiff && (
                    <span className="ml-1.5 inline-block h-1.5 w-1.5 rounded-full bg-amber-400 align-middle" title="Values differ" />
                  )}
                </td>
                {firms.map((firm) => (
                  <td
                    key={firm.id}
                    className={`px-4 py-3 text-sm text-center ${
                      isDiff ? 'text-white font-medium' : 'text-white/60'
                    }`}
                  >
                    {row.format ? row.format(firm) : String(firm[row.key as keyof ScoredFirm] ?? '—')}
                  </td>
                ))}
              </tr>
            );
          })}
        </tbody>
        <tfoot>
          <tr>
            <td className="px-4 py-4" />
            {firms.map((firm) => (
              <td key={firm.id} className="px-4 py-4 text-center">
                <div className="flex flex-col gap-2 items-center">
                  <Button asChild size="sm" className="bg-[#00D4AA] hover:bg-[#00D4AA]/90 text-[#08080F] font-semibold text-xs w-full max-w-[140px]">
                    <Link href={`/firms/${firm.slug}`}>View Full Details</Link>
                  </Button>
                  <Button asChild size="sm" variant="outline" className="border-white/15 text-white/60 hover:text-white text-xs w-full max-w-[140px]">
                    <a href={firm.affiliateUrl ?? firm.websiteUrl ?? '#'} target="_blank" rel="noopener noreferrer">
                      <ExternalLink className="h-3 w-3 mr-1" /> Visit Site
                    </a>
                  </Button>
                </div>
              </td>
            ))}
          </tr>
        </tfoot>
      </table>
    </div>
  );
}
```
